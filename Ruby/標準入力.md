# 標準入力受け取り方  

## 文字列を受け取る  gets.chomp  
name = gets.chomp  
puts "Hello #{name}"  

world -> Hello world  

## 整数を受け取る  gets.to_i  
number = gets.to_i  
puts number * 10  

12 -> 120  

## 繰り返し回数を受け取る  
greeting = "Hello world"  
count = gets.to_i  

count.times do  
　puts greeting  
end  

## 複数のデータを受け取る  
count = gets.to_i  

count.times do  
　name = gets.chomp  
　puts "Hello #{name}"  
end
